Requires:

  - Node.js https://nodejs.org/en/download/
  
  - Yarn https://yarnpkg.com/lang/en/docs/install/
    
  - Nodemon "npm install -g nodemon" https://www.npmjs.com/package/nodemon


To run node.js and react:

    > git pull
    
    > cd client
    
    > yarn install
    
    > cd ..
    
    > yarn install
    
    > yarn dev
